#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#define N 6
void find_two_largest(int a[], int n, int* largest, int* second_largest) {
    // 直接修改指针指向的内容
    *largest = a[0];
    *second_largest = a[0];

    // 找到最大值
    for (int i = 1; i < n; i++) {
        if (a[i] > *largest) {
            *largest = a[i];
        }
    }

    // 找到次大值（不能等于最大值）
    for (int i = 0; i < n; i++) {
        if (a[i] > *second_largest && a[i] != *largest) {
            *second_largest = a[i];
        }
    }
}
int main() {
	int a[N] = { 0 };
	for(int i = 1; i < N; i++) {
		scanf("%d", &a[i]);
	}
	int l1 , l2;
	find_two_largest(a, N, &l1, &l2);
	printf("Largest: %d\nSecond Largest: %d\n", l1, l2);
	return 0;
}

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

void split_date(int day_of_year, int year, int* month, int* day)
{
    int days_in_month[] = { 31, 28, 31, 30, 31, 30,
                          31, 31, 30, 31, 30, 31 };

    // 检查闰年
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
        days_in_month[1] = 29;
    }

    int remaining_days = day_of_year;

    for (int i = 0; i < 12; i++) {
        if (remaining_days <= days_in_month[i]) {
            *month = i + 1;      // 月份从1开始
            *day = remaining_days;
            return;
        }
        remaining_days -= days_in_month[i];
    }

    // 如果day_of_year超过365/366天
    *month = 0;
    *day = 0;
}

int main() {
    int day_of_year = 182;  // 2024年的第182天
    int year = 2024;
    int month, day;

    split_date(day_of_year, year, &month, &day);

    printf("Year: %d\n", year);
    printf("Day of year: %d\n", day_of_year);
    printf("Month: %d, Day: %d\n", month, day);
    // 输出应该是: Month: 6, Day: 30

    return 0;
}

{
	int* largest = &a[0];
	for (int i = 1; i < n; i++) {
		if (a[i] > *largest) {
			largest = &a[i];

void reduce(int numerator, int denominator,

int main() {

}
void reduce(int numerator, int denominator,
	int* reduced_numerator,
	int* reduced_denominator)
{
	int gcd = 1;
	for (int i = 1; i <= numerator && i <= denominator; i++) {
		if (numerator % i == 0 && denominator % i == 0) {
			gcd = i;
		}
	}
	*reduced_numerator = numerator / gcd;
	*reduced_denominator = denominator / gcd;
}
