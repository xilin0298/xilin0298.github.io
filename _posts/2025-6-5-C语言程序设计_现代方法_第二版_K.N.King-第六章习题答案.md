### 1
去除if语句的分号，变量n不能再printf函数里还是以变量显示
```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main() {
	/* 编写程序，找出用户输入的一串数中的最大数。程序需要提示用户一个一个地输入数。当用户输入
0 或负数时，程序必须显示出已输入的最大非负数 */
	printf("请输入一个数，输入0或负数结束：\n");
	float num;
	float max = 0;
	scanf("%f", &num);
	while (1) {
		if(num <= 0) {
			break;
		}
		if(num > max) {
			max = num;
		}
		scanf("%f", &num);
	}
	printf("最大非负数是：%f\n", max);	
}
```

```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main() {
	/*编写程序，要求用户输入两个整数，然后计算并显示这两个整数的最大公约数（GCD）：*/
	printf("请输入两个整数：");
	int m,n;
	scanf("%d %d", &m, &n);
	while (1) {
		if (n == 0) {
			printf("最大公约数是：%d\n", m);
			return 0;
		}
		else {
			int remainder = m % n;
			m = n;
			n = remainder;
		}
	}
}
```

			m = n;

		}

	int n1 = n0 / gcd;

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main() {
	/*编写程序，要求用户输入一个分数，然后将其约分为最简分式*/
	float commission, value;
	while (1) {
		printf("Enter value of trade: ");
		scanf("%f", &value);
		if (value == 0.0f) {
			break;
		}
		else {
			if (value < 2500.00f)
				commission = 30.00f + .017f * value;
			else if (value < 6250.00f)
				commission = 56.00f + .0066f * value;
			else if (value < 20000.00f)
				commission = 76.00f + .0034f * value;
			else if (value < 50000.00f)
				commission = 100.00f + .0022f * value;
			else if (value < 500000.00f)
				commission = 155.00f + .0011f * value;
			else
				commission = 255.00f + .0009f * value;
			if (commission < 39.00f)
				commission = 39.00f;
			printf("Commission: $%.2f\n", commission);
			continue;
		}
	}
	
		
		return 0;
	}
